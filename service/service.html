Multiplication Table

// define the size of multiplication table
const size = 10;
// outer loop
for (let i = 1; i <- size; i++) {
  // inner loop
  for (let j = 1; j <= size; j++ ){
    // calculate and display the product
    const product = i * j;
    console.log(`${i} * ${j} = ${product}`)
  }
}
